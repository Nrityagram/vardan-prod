<script type="text/javascript">
  const modal = document.querySelector('#donate-options');
  const openModal = document.querySelector('.open-modal');
  const closeModal = document.querySelector('.close-button');

  function clearOptions() {
    // Clear any previous selection
    const allOptions = document.querySelectorAll('.citizen-type');
    allOptions.forEach((option) => {
      option.checked = false;
    });
  }

  openModal.addEventListener('click', () => {
    clearOptions();
    modal.showModal();
  });

  closeModal.addEventListener('click', () => {
    modal.close();
  });

  const allCitizenOptions = document.querySelectorAll('.citizen-type');
  allCitizenOptions.forEach((targetOption) => {
    targetOption.addEventListener('click', () => {
      // Load appropriate donation page based
      // on the citizenship choice made
      switch (targetOption.id) {
        case 'india':
          location.href = "/makeadonation";
          break;
        case 'nri':
          location.href = "/makeadonation";
          break;
        case 'usa':
          location.href = "/makeadonation-usa";
          break;
        case 'othercountry':
          location.href = "/makeadonation-other";
          break;
        default:
          break;
      }
    });
  });
</script>
